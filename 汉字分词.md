```javascript
  function tokenizeJA(text) {
    var it = Intl.v8BreakIterator(['ja-JP'], {type:'word'})
    it.adoptText(text)
    var words = []
  
    var cur = 0, prev = 0
  
    while (cur < text.length) {
      prev = cur
      cur = it.next()
      words.push(text.substring(prev, cur))
    }
  
    return words
  
  }
  
  console.log(tokenizeJA("我们的移动设备和各种后端语言都支持相同的数据。"))
  ```

const str = "我认为空间对于学习速度和表达方式也很有用。";
